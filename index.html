tcpdump

capture: 
sudo tcpdump -i en0 -s 0 -B 524288 -w dump.pcap
or
sudo tcpdump -Ii en0 -vA -w dump.pcap
or
sudo tcpdump -i en0 -C 1 -z gzip -w ~/tests/tcpdump/`date "+%Y-%m-%d_%H:%M:%S"'+%s'`.pcap
read:
tcpdump -s 0 -n -e -x -vvv -r dump.pcap
or
tcpdump -nr data.pca
